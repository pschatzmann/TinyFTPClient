218  bool checkResult(const char *expected[], const char *command,
219  bool wait_for_data = true) {
220  // consume all result lines
221  bool ok = false;
222  result_reply[0] = '\0';
223  Stream *stream_ptr = command_ptr;
224 
225  char result_str[FTP_RESULT_BUFFER_SIZE];
226  if (wait_for_data || stream_ptr->available()) {
227  // wait for reply
228  while (stream_ptr->available() == 0) {
229  delay(100);
230  }
231  // read reply
232  // result_str = stream_ptr->readStringUntil('\n').c_str();
233  CStringFunctions::readln(*stream_ptr, result_str, FTP_RESULT_BUFFER_SIZE);
234 
235  if (strlen(result_str) > 3) {
236  FTPLogger::writeLog(LOG_DEBUG, "FTPBasicAPI::checkResult", result_str);
237  strncpy(result_reply, result_str, sizeof(result_reply) - 1);
238  // if we did not expect anything
239  if (expected[0] == nullptr) {
240  ok = true;
241  FTPLogger::writeLog(LOG_DEBUG, "FTPBasicAPI::checkResult",
242  "success because of not expected result codes");
243  } else {
244  // check for valid codes
245  char msg[80];
246  for (int j = 0; expected[j] != nullptr; j++) {
247  sprintf(msg, "- checking with %s", expected[j]);
248  FTPLogger::writeLog(LOG_DEBUG, "FTPBasicAPI::checkResult", msg);
249  if (strncmp(result_str, expected[j], 3) == 0) {
250  sprintf(msg, " -> success with %s", expected[j]);
251  FTPLogger::writeLog(LOG_DEBUG, "FTPBasicAPI::checkResult", msg);
252  ok = true;
253  break;
254  }
255  }
256  }
257  } else {
258  // if we got am empty line and we dont need to wait we are still ok
259  if (!wait_for_data) ok = true;
260  }
261  } else {
262  ok = true;
263  }
264 
265  // log error
266  if (!ok) {
267  FTPLogger::writeLog(LOG_ERROR, "FTPBasicAPI::checkResult", command);
268  FTPLogger::writeLog(LOG_ERROR, "FTPBasicAPI::checkResult", result_reply);
269  }
270  return ok;
271  }
