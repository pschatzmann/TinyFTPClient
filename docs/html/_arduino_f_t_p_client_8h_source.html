136 // occurence of the character
137 for (int i = 0; i < strlen(str); i++) {

147 static int readln(Stream &stream, char* str, int maxLen) {
148 int len = 0;
149 if (maxLen>stream.available()){
150 maxLen = stream.available();
151 }
152 for (int j=0;j<maxLen;j++){
153 len = j;
154 char c = stream.read();
155 if (c==0 || c=='\n'){
156 break;
157 }
158 str[j] = c;
159 }
160 memset(str+len,0,maxLen-len);
161 return len;
162 }
