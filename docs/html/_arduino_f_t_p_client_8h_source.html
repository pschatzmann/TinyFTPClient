84  static int readln(Stream &stream, char *str, int maxLen) {
85  int len = 0;
86  if (maxLen > stream.available()) {
87  maxLen = stream.available();
88  }
89  for (int j = 0; j < maxLen; j++) {
90  len = j;
91  char c = stream.read();
92  if (c == 0 || c == '\n') {
93  break;
94  }
95  str[j] = c;
96  }
97  memset(str + len, 0, maxLen - len);
98  return len;
99  }
